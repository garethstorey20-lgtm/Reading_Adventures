  <h1>PONG</h1>
  <div class="info">Mouse controls left paddle • Hit power-ups with the ball • First to 6 wins</div>

<script>
(() => {
  const canvas = document.getElementById('game');
  const ctx = canvas.getContext('2d');
  const overlay = document.getElementById('overlay');
  const overlayTitle = document.getElementById('overlayTitle');
  const overlayText = document.getElementById('overlayText');
  const startBtn = document.getElementById('startBtn');
  const hitsDisplay = document.getElementById('hitsDisplay');
  const speedDisplay = document.getElementById('speedDisplay');
  const ballsDisplay = document.getElementById('ballsDisplay');

  const W = 800, H = 500;

  function resize() {
    const rect = canvas.getBoundingClientRect();
    const dpr = window.devicePixelRatio || 1;
    canvas.width = rect.width * dpr;
    canvas.height = rect.height * dpr;
    ctx.setTransform(dpr * (rect.width / W), 0, 0, dpr * (rect.height / H), 0, 0);
  }
  window.addEventListener('resize', resize);

  // Audio
  let audioCtx = null;
  function ensureAudio() {
    if (!audioCtx) {
      try { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } catch(e) {}
    }
    if (audioCtx && audioCtx.state === 'suspended') audioCtx.resume();
  }
  function beep(freq, duration = 0.08, type = 'square', vol = 0.15) {
    if (!audioCtx) return;
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = type;
    osc.frequency.value = freq;
    gain.gain.setValueAtTime(vol, audioCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
    osc.connect(gain);
    gain.connect(audioCtx.destination);
    osc.start();
    osc.stop(audioCtx.currentTime + duration);
  }
  function powerUpSound(type) {
    if (!audioCtx) return;
    if (type === 'split') {
      beep(700, 0.08, 'triangle', 0.18);
      setTimeout(() => beep(900, 0.08, 'triangle', 0.18), 70);
    } else if (type === 'red') {
      beep(180, 0.15, 'sawtooth', 0.2);
      setTimeout(() => beep(120, 0.2, 'sawtooth', 0.2), 100);
    } else if (type === 'blue') {
      beep(600, 0.15, 'sine', 0.18);
      setTimeout(() => beep(400, 0.2, 'sine', 0.18), 120);
    }
  }

  // Constants
  const PADDLE_W = 12, PADDLE_H = 90;
  const BALL_SIZE = 12;
  const WIN_SCORE = 6;
  const BASE_BALL_SPEED = 5.5;
  const BASE_AI_SPEED = 3;
  const AI_REACT_THRESHOLD = W * 0.45;
  const MAX_BALLS = 16;
  const POWERUP_GRACE_MS = 2000; // no power-ups for 2s after each play start

  // State
  let player = { x: 20, y: H/2 - PADDLE_H/2, w: PADDLE_W, h: PADDLE_H };
  let ai = { x: W - 20 - PADDLE_W, y: H/2 - PADDLE_H/2, w: PADDLE_W, h: PADDLE_H };
  let balls = [];
  let powerUps = [];
  let scores = { player: 0, ai: 0 };
  let hits = 0;
  let speedMultiplier = 1;
  let running = false;
  let gameOver = false;
  let mouseY = H / 2;
  let playStartAt = 0;            // timestamp of current play start
  let nextPowerUpAt = 0;
  let aiNoiseSeed = Math.random() * 1000;
  let respawnScheduled = false;

  function makeBall(x, y, vx, vy, opts = {}) {
    return {
      x, y, vx, vy,
      size: opts.size || BALL_SIZE,
      effect: opts.effect || null,
      slowUntil: opts.slowUntil || 0,
      preSlowSpeed: opts.preSlowSpeed || 0,
      ghostSide: opts.ghostSide || null,
      trail: []
    };
  }

  function spawnBall(dir = Math.random() > 0.5 ? 1 : -1) {
    const angle = (Math.random() * 0.6 - 0.3);
    const speed = BASE_BALL_SPEED * speedMultiplier;
    balls.push(makeBall(W/2, H/2, Math.cos(angle) * speed * dir, Math.sin(angle) * speed));
    // Reset power-up grace period for this play
    const now = performance.now();
    playStartAt = now;
    nextPowerUpAt = now + POWERUP_GRACE_MS + Math.random() * 1500;
    // Clear any leftover power-ups from the previous play
    powerUps = [];
  }

  function resetGame() {
    scores.player = 0;
    scores.ai = 0;
    hits = 0;
    speedMultiplier = 1;
    gameOver = false;
    balls = [];
    powerUps = [];
    respawnScheduled = false;
    spawnBall();
    updateStats();
  }

  function updateStats() {
    hitsDisplay.textContent = `Hits: ${hits}`;
    speedDisplay.textContent = `Speed: ${Math.round(speedMultiplier * 100)}%`;
    ballsDisplay.textContent = `Balls: ${balls.length}`;
  }

  // Input
  canvas.addEventListener('mousemove', (e) => {
    const rect = canvas.getBoundingClientRect();
    const scale = H / rect.height;
    mouseY = (e.clientY - rect.top) * scale;
  });
  canvas.addEventListener('touchmove', (e) => {
    e.preventDefault();
    const rect = canvas.getBoundingClientRect();
    const scale = H / rect.height;
    mouseY = (e.touches[0].clientY - rect.top) * scale;
  }, { passive: false });

  function rectsOverlap(a, b) {
    return a.x < b.x + b.w && a.x + a.w > b.x && a.y < b.y + b.h && a.y + a.h > b.y;
  }

  // Power-ups
  const POWERUP_TYPES = ['split', 'red', 'blue'];
  const POWERUP_COLORS = {
    split:  '#ffdd33',
    red:    '#ff3355',
    blue:   '#33aaff'
  };
  const POWERUP_SYMBOL = {
    split:  '2',
    red:    'R',
    blue:   'S'
  };

  function spawnPowerUp() {
    if (powerUps.length >= 2) return;
    const type = POWERUP_TYPES[Math.floor(Math.random() * POWERUP_TYPES.length)];
    const x = W * 0.3 + Math.random() * W * 0.4;
    const y = H * 0.15 + Math.random() * H * 0.7;
    powerUps.push({
      x, y,
      r: 18,
      type,
      bornAt: performance.now(),
      lifetime: 12000
    });
  }

  function applyPowerUp(ball, type) {
    powerUpSound(type);
    if (type === 'split') {
      if (balls.length >= MAX_BALLS) return;
      const speed = Math.hypot(ball.vx, ball.vy);
      const angle = Math.atan2(ball.vy, ball.vx);
      const spread = 0.35;
      ball.vx = Math.cos(angle - spread) * speed;
      ball.vy = Math.sin(angle - spread) * speed;
      balls.push(makeBall(ball.x, ball.y,
        Math.cos(angle + spread) * speed,
        Math.sin(angle + spread) * speed));
    } else if (type === 'red') {
      ball.effect = 'red';
      ball.ghostSide = ball.vx > 0 ? 'right' : 'left';
    } else if (type === 'blue') {
      ball.effect = 'blue';
      const now = performance.now();
      const speed = Math.hypot(ball.vx, ball.vy);
      ball.preSlowSpeed = speed;
      const factor = 0.4;
      ball.vx *= factor;
      ball.vy *= factor;
      ball.slowUntil = now + 5000;
    }
  }

  function updateAI() {
    const aiCenter = ai.y + PADDLE_H / 2;

    let target = null, bestX = -Infinity;
    for (const b of balls) {
      if (b.vx > 0 && b.x > bestX) { target = b; bestX = b.x; }
    }
    if (!target) {
      const centerDiff = H/2 - aiCenter;
      ai.y += Math.sign(centerDiff) * Math.min(Math.abs(centerDiff), 1.5);
      ai.y = Math.max(0, Math.min(H - PADDLE_H, ai.y));
      return;
    }

    const wobble = Math.sin(performance.now() / 350 + aiNoiseSeed) * 35;
    const targetY = target.y + wobble;
    const aiSpeed = BASE_AI_SPEED * Math.min(speedMultiplier, 1.4);

    let reactiveness = 1;
    if (target.x < AI_REACT_THRESHOLD) reactiveness = 0.35;

    const diff = targetY - aiCenter;
    const move = Math.sign(diff) * Math.min(Math.abs(diff), aiSpeed) * reactiveness;
    ai.y += move;
    ai.y = Math.max(0, Math.min(H - PADDLE_H, ai.y));
  }

  function update(now) {
    if (!running || gameOver) return;

    const targetY = mouseY - PADDLE_H / 2;
    player.y += (targetY - player.y) * 0.35;
    player.y = Math.max(0, Math.min(H - PADDLE_H, player.y));

    updateAI();

    // Power-ups can only spawn after grace period has elapsed
    if (now - playStartAt >= POWERUP_GRACE_MS && now >= nextPowerUpAt) {
      spawnPowerUp();
      nextPowerUpAt = now + 4000 + Math.random() * 3000;
    }
    powerUps = powerUps.filter(p => now - p.bornAt < p.lifetime);

    for (let i = balls.length - 1; i >= 0; i--) {
      const ball = balls[i];

      if (ball.effect === 'blue' && now > ball.slowUntil) {
        const currentSpeed = Math.hypot(ball.vx, ball.vy);
        const target = ball.preSlowSpeed || (BASE_BALL_SPEED * speedMultiplier);
        if (currentSpeed > 0) {
          const r = target / currentSpeed;
          ball.vx *= r;
          ball.vy *= r;
        }
        ball.effect = null;
        beep(800, 0.1, 'sine', 0.15);
      }

      ball.trail.push({ x: ball.x, y: ball.y });
      if (ball.trail.length > 8) ball.trail.shift();

      ball.x += ball.vx;
      ball.y += ball.vy;

      let wallHit = false;
      if (ball.y <= 0) {
        ball.y = 0;
        ball.vy = -ball.vy;
        wallHit = true;
      } else if (ball.y + ball.size >= H) {
        ball.y = H - ball.size;
        ball.vy = -ball.vy;
        wallHit = true;
      }
      if (wallHit) {
        beep(220, 0.04, 'square', 0.08);
      }

      const ballRect = { x: ball.x, y: ball.y, w: ball.size, h: ball.size };

      // Player paddle
      if (ball.vx < 0 && rectsOverlap(ballRect, player)) {
        if (!(ball.effect === 'red' && ball.ghostSide === 'left')) {
          ball.x = player.x + player.w;
          const hitPos = (ball.y + ball.size/2 - (player.y + PADDLE_H/2)) / (PADDLE_H/2);
          const speed = Math.hypot(ball.vx, ball.vy);
          const angle = hitPos * (Math.PI / 4);
          ball.vx = Math.abs(Math.cos(angle) * speed);
          ball.vy = Math.sin(angle) * speed;
          onPaddleHit(true);
          if (ball.effect === 'red') { ball.effect = null; ball.ghostSide = null; }
        }
      }

      // AI paddle
      if (ball.vx > 0 && rectsOverlap(ballRect, ai)) {
        if (!(ball.effect === 'red' && ball.ghostSide === 'right')) {
          ball.x = ai.x - ball.size;
          const hitPos = (ball.y + ball.size/2 - (ai.y + PADDLE_H/2)) / (PADDLE_H/2);
          const speed = Math.hypot(ball.vx, ball.vy);
          const angle = hitPos * (Math.PI / 4);
          ball.vx = -Math.abs(Math.cos(angle) * speed);
          ball.vy = Math.sin(angle) * speed;
          onPaddleHit(false);
          if (ball.effect === 'red') { ball.effect = null; ball.ghostSide = null; }
        }
      }

      // Power-up collision
      for (let j = powerUps.length - 1; j >= 0; j--) {
        const p = powerUps[j];
        const cx = ball.x + ball.size/2;
        const cy = ball.y + ball.size/2;
        const dx = cx - p.x, dy = cy - p.y;
        if (dx*dx + dy*dy < (p.r + ball.size/2) * (p.r + ball.size/2)) {
          applyPowerUp(ball, p.type);
          powerUps.splice(j, 1);
        }
      }

      // Score / off-screen
      if (ball.x + ball.size < 0) {
        scores.ai++;
        beep(150, 0.25, 'sawtooth', 0.15);
        balls.splice(i, 1);
      } else if (ball.x > W) {
        scores.player++;
        beep(660, 0.18, 'triangle', 0.15);
        balls.splice(i, 1);
      }
    }

    // Round end / respawn
    if (balls.length === 0 && !respawnScheduled) {
      if (!checkWin()) {
        respawnScheduled = true;
        setTimeout(() => {
          respawnScheduled = false;
          if (running && !gameOver && balls.length === 0) {
            spawnBall();
          }
        }, 600);
      }
    }

    updateStats();
  }

  function onPaddleHit(isPlayer) {
    hits++;
    beep(isPlayer ? 480 : 380, 0.07, 'square', 0.18);
    if (hits % 10 === 0) {
      speedMultiplier *= 1.1;
      for (const b of balls) {
        if (b.effect === 'blue') {
          b.preSlowSpeed *= 1.1;
        } else {
          b.vx *= 1.1;
          b.vy *= 1.1;
        }
      }
      beep(880, 0.1, 'triangle', 0.15);
      setTimeout(() => beep(1100, 0.1, 'triangle', 0.15), 100);
    }
  }

  function checkWin() {
    if (scores.player >= WIN_SCORE || scores.ai >= WIN_SCORE) {
      gameOver = true;
      running = false;
      const won = scores.player >= WIN_SCORE;
      overlayTitle.textContent = won ? 'YOU WIN!' : 'YOU LOSE';
      overlayTitle.style.color = won ? '#00ff88' : '#ff4466';
      overlayTitle.style.textShadow = won ? '0 0 15px #00ff88' : '0 0 15px #ff4466';
      overlayText.innerHTML = `Final Score: <strong>${scores.player} - ${scores.ai}</strong><br>Total Hits: ${hits} • Final Speed: ${Math.round(speedMultiplier * 100)}%`;
      startBtn.textContent = 'PLAY AGAIN';
      overlay.classList.remove('hidden');
      if (won) {
        beep(523, 0.15, 'triangle', 0.2);
        setTimeout(() => beep(659, 0.15, 'triangle', 0.2), 150);
        setTimeout(() => beep(784, 0.3, 'triangle', 0.2), 300);
      } else {
        beep(300, 0.2, 'sawtooth', 0.2);
        setTimeout(() => beep(200, 0.4, 'sawtooth', 0.2), 200);
      }
      return true;
    }
    return false;
  }

  function ballColor(ball) {
    if (ball.effect === 'red') return '#ff3355';
    if (ball.effect === 'blue') return '#33aaff';
    return '#ffffff';
  }

  function draw(now) {
    ctx.fillStyle = '#000';
    ctx.fillRect(0, 0, W, H);

    ctx.strokeStyle = '#00ff88';
    ctx.globalAlpha = 0.3;
    ctx.lineWidth = 2;
    ctx.setLineDash([10, 12]);
    ctx.beginPath();
    ctx.moveTo(W/2, 0);
    ctx.lineTo(W/2, H);
    ctx.stroke();
    ctx.setLineDash([]);
    ctx.globalAlpha = 1;

    ctx.fillStyle = '#00ff88';
    ctx.font = 'bold 60px Courier New';
    ctx.textAlign = 'center';
    ctx.globalAlpha = 0.55;
    ctx.fillText(scores.player, W/2 - 60, 70);
    ctx.fillText(scores.ai, W/2 + 60, 70);
    ctx.globalAlpha = 1;

    for (const p of powerUps) {
      const age = now - p.bornAt;
      const remaining = p.lifetime - age;
      const fadeIn = Math.min(1, age / 300);
      const fadeOut = remaining < 2000 ? Math.max(0.2, remaining / 2000) : 1;
      const alpha = fadeIn * fadeOut;
      const pulse = 1 + Math.sin(now / 200) * 0.08;
      const color = POWERUP_COLORS[p.type];

      ctx.globalAlpha = alpha * 0.3;
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.r * pulse * 1.6, 0, Math.PI * 2);
      ctx.fillStyle = color;
      ctx.fill();

      ctx.globalAlpha = alpha;
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.r * pulse, 0, Math.PI * 2);
      ctx.fillStyle = color;
      ctx.shadowColor = color;
      ctx.shadowBlur = 15;
      ctx.fill();
      ctx.shadowBlur = 0;

      ctx.fillStyle = '#000';
      ctx.font = 'bold 20px Courier New';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.fillText(POWERUP_SYMBOL[p.type], p.x, p.y + 1);
      ctx.textBaseline = 'alphabetic';
      ctx.globalAlpha = 1;
    }

    ctx.fillStyle = '#00ff88';
    ctx.shadowColor = '#00ff88';
    ctx.shadowBlur = 10;
    ctx.fillRect(player.x, player.y, player.w, player.h);
    ctx.fillRect(ai.x, ai.y, ai.w, ai.h);
    ctx.shadowBlur = 0;

    for (const ball of balls) {
      const color = ballColor(ball);

      for (let i = 0; i < ball.trail.length; i++) {
        const t = ball.trail[i];
        ctx.globalAlpha = (i / ball.trail.length) * 0.4;
        ctx.fillStyle = color;
        ctx.fillRect(t.x, t.y, ball.size, ball.size);
      }
      ctx.globalAlpha = 1;
      ctx.fillStyle = color;
      ctx.shadowColor = color;
      ctx.shadowBlur = 15;
      ctx.fillRect(ball.x, ball.y, ball.size, ball.size);
      ctx.shadowBlur = 0;

      if (ball.effect === 'blue') {
        const remaining = Math.max(0, ball.slowUntil - now);
        const frac = remaining / 5000;
        ctx.strokeStyle = '#33aaff';
        ctx.lineWidth = 2;
        ctx.beginPath();
        ctx.arc(ball.x + ball.size/2, ball.y + ball.size/2, ball.size, -Math.PI/2, -Math.PI/2 + Math.PI*2*frac);
        ctx.stroke();
      }
    }
  }

  function loop(ts) {
    update(ts);
    draw(ts);
    requestAnimationFrame(loop);
  }

  startBtn.addEventListener('click', () => {
    ensureAudio();
    if (gameOver || balls.length === 0) {
      resetGame();
    }
    overlay.classList.add('hidden');
    running = true;
  });

  resize();
  resetGame();
  running = false;
  requestAnimationFrame(loop);
})();
